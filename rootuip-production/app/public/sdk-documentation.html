Quick Start

Initialize the SDK and start tracking containers in minutes

Basic Setup 
// Import the SDK
const { RootuipClient } = require('@rootuip/sdk');

// Initialize the client
const client = new RootuipClient({
    apiKey: 'your_api_key_here',
    environment: 'production', // or 'sandbox'
    timeout: 30000,
    retries: 3
});

// Track a container
async function trackContainer() {
    try {
        const container = await client.containers.track({
            containerNumber: 'MSKU1234567',
            carrier: 'maersk',
            referenceNumber: 'BOOK123456'
        });
        
        console.log('Container tracked:', container.id);
        console.log('Current status:', container.status);
        console.log('Location:', container.currentLocation);
        
        return container;
    } catch (error) {
        console.error('Error tracking container:', error);
    }
}

// Start tracking
trackContainer();

Container Tracking

Complete guide to tracking containers across all supported carriers

track(options)

client.containers.track(options: TrackingOptions) → Promise<Container>

Start tracking a container by providing container details. The system will automatically detect the carrier if not specified.

Parameters

Parameter Type Required Description
containerNumber string Required The container identification number (11 characters)
carrier string Optional Carrier code (auto-detected if not provided)
referenceNumber string Optional Booking or Bill of Lading reference
webhookUrl string Optional URL to receive real-time updates
customFields object Optional Additional metadata to associate with the container

Advanced Tracking Example 
// Track multiple containers with real-time updates
async function trackMultipleContainers() {
    const containers = [
        { containerNumber: 'MSKU1234567', carrier: 'maersk' },
        { containerNumber: 'MSCU9876543', carrier: 'msc' },
        { containerNumber: 'CMAU5555555', carrier: 'cma-cgm' }
    ];
    
    // Track all containers
    const trackingPromises = containers.map(container => 
        client.containers.track({
            ...container,
            webhookUrl: 'https://your-app.com/webhooks/container-updates',
            customFields: {
                department: 'logistics',
                priority: 'high'
            }
        })
    );
    
    try {
        const trackedContainers = await Promise.allSettled(trackingPromises);
        
        trackedContainers.forEach((result, index) => {
            if (result.status === 'fulfilled') {
                console.log(`✅ ${containers[index].containerNumber}: ${result.value.id}`);
            } else {
                console.error(`❌ ${containers[index].containerNumber}: ${result.reason.message}`);
            }
        });
        
        return trackedContainers;
    } catch (error) {
        console.error('Batch tracking failed:', error);
    }
}

// Set up real-time subscriptions
function subscribeToUpdates(containerId) {
    return client.containers.subscribe(containerId, {
        events: ['location_update', 'status_change', 'demurrage_risk'],
        onUpdate: (update) => {
            console.log('Container update:', update);
            
            // Handle different event types
            switch (update.event) {
                case 'demurrage_risk':
                    if (update.data.level === 'high') {
                        sendAlert(`HIGH D&D RISK: ${update.containerNumber}`);
                    }
                    break;
                    
                case 'arrived_at_destination':
                    notifyTeam(`Container ${update.containerNumber} has arrived!`);
                    break;
            }
        },
        onError: (error) => {
            console.error('Subscription error:', error);
        }
    });
}

// Verify webhook signature
function verifyWebhookSignature(payload, signature, secret) {
    const crypto = require('crypto');
    const expectedSignature = crypto
        .createHmac('sha256', secret)
        .update(payload)
        .digest('hex');
    
    return `sha256=${expectedSignature}` === signature;
}

Error Handling

Robust error handling and retry strategies

⚠️ Important

Always implement proper error handling when working with external APIs. The SDK provides detailed error information to help you debug issues.

Error Handling Best Practices 
// Comprehensive error handling
async function trackContainerWithErrorHandling(containerNumber) {
    try {
        const container = await client.containers.track({
            containerNumber,
            carrier: 'auto-detect'
        });
        
        return container;
        
    } catch (error) {
        // Handle specific error types
        switch (error.code) {
            case 'INVALID_CONTAINER_NUMBER':
                console.error('Invalid container format:', error.message);
                throw new Error('Please check the container number format');
                
            case 'CARRIER_NOT_SUPPORTED':
                console.error('Carrier not supported:', error.details.carrier);
                // Try alternative tracking method
                return await tryAlternativeTracking(containerNumber);
                
            case 'RATE_LIMIT_EXCEEDED':
                console.warn('Rate limit hit, retrying after delay...');
                await new Promise(resolve => setTimeout(resolve, error.retryAfter * 1000));
                return await trackContainerWithErrorHandling(containerNumber);
                
            case 'AUTHENTICATION_FAILED':
                console.error('API key invalid or expired');
                throw new Error('Please check your API credentials');
                
            case 'NETWORK_ERROR':
                console.error('Network issue:', error.message);
                // Implement exponential backoff
                return await retryWithBackoff(() => 
                    client.containers.track({ containerNumber })
                );
                
            default:
                console.error('Unexpected error:', error);
                throw error;
        }
    }
}

// Retry with exponential backoff
async function retryWithBackoff(fn, maxRetries = 3, baseDelay = 1000) {
    for (let attempt = 1; attempt <= maxRetries; attempt++) {
        try {
            return await fn();
        } catch (error) {
            if (attempt === maxRetries) throw error;
            
            const delay = baseDelay * Math.pow(2, attempt - 1);
            console.log(`Retry ${attempt}/${maxRetries} after ${delay}ms`);
            await new Promise(resolve => setTimeout(resolve, delay));
        }
    }
}

Testing

Test your integration with mock clients and test utilities

Unit Testing Example 
// Import test utilities
const { MockRootuipClient, createTestContainer } = require('@rootuip/sdk/testing');

describe('Container Tracking Service', () => {
    let client;
    let trackingService;
    
    beforeEach(() => {
        // Use mock client for testing
        client = new MockRootuipClient();
        trackingService = new TrackingService(client);
    });
    
    test('should track container successfully', async () => {
        // Setup mock response
        const mockContainer = createTestContainer({
            containerNumber: 'MSKU1234567',
            status: 'in_transit',
            currentLocation: {
                port: 'CNSHA',
                portName: 'Shanghai, China'
            }
        });
        
        client.containers.track.mockResolvedValue(mockContainer);
        
        // Test the service
        const result = await trackingService.track('MSKU1234567');
        
        expect(result.containerNumber).toBe('MSKU1234567');
        expect(result.status).toBe('in_transit');
        expect(client.containers.track).toHaveBeenCalledWith({
            containerNumber: 'MSKU1234567',
            carrier: 'auto-detect'
        });
    });
    
    test('should handle tracking errors gracefully', async () => {
        // Setup mock error
        client.containers.track.mockRejectedValue(
            new Error('Invalid container number')
        );
        
        // Test error handling
        await expect(
            trackingService.track('INVALID123')
        ).rejects.toThrow('Invalid container number');
    });
    
    test('should setup webhook correctly', async () => {
        const webhookConfig = {
            url: 'https://test.com/webhook',
            events: ['container.status_changed']
        };
        
        client.webhooks.create.mockResolvedValue({
            id: 'webhook_123',
            ...webhookConfig
        });
        
        const webhook = await trackingService.setupWebhook(webhookConfig);
        
        expect(webhook.id).toBe('webhook_123');
        expect(client.webhooks.create).toHaveBeenCalledWith(webhookConfig);
    });
});
